\section{Contexte et objectifs}

\begin{frame}{Contexte : un marché vie en pleine croissance}
  \vspace{0.2cm}
  \begin{columns}[T]
    \begin{column}{0.60\textwidth}
      \begin{itemize}\itemsep0.25cm
        \item zone FANAF 2021 : 1\,488,6 Mds FCFA de primes, la vie
              (549,6 Mds) croissant plus vite que le marché (+13,6\,\%) ;
        \item au Sénégal, la vie accélère : 99,0 Mds de primes en 2023,
              113,6 Mds en 2024, soit 38\,\% du marché national ;
        \item indemnités de fin de carrière obligatoires (CCNI), à provisionner
              (SYSCOHADA révisé, IAS 19) : l'externalisation auprès d'un
              assureur vie s'impose aux entreprises.
      \end{itemize}
    \end{column}
    \begin{column}{0.36\textwidth}
      \stat{113,6 Mds}{de primes vie au Sénégal en 2024 (+14,8\,\%)}
      \vspace{0.25cm}
      \stat{346,1 Mds}{de provisions mathématiques vie fin 2024 (+16,4\,\%)}
    \end{column}
  \end{columns}
  \vspace{0.1cm}
  {\tiny Sources : FANAF, annuaire des sociétés membres 2026 (exercices 2023-2024) ; FANAF, Le marché de l'assurance en Afrique, données 2017 à 2021.}
\end{frame}

\begin{frame}{Objectifs du projet}
  \vspace{0.2cm}
  \textbf{Énoncé :} mener une enquête auprès des institutions d'assurance afin
  de recueillir les produits commercialisés, puis produire la note technique du
  produit le plus commercialisé.
  \vspace{0.3cm}
  \begin{itemize}\itemsep0.25cm
    \item identifier le produit vie phare du marché et justifier ce choix par
          les données de marché (FANAF), une enquête auprès des compagnies
          étant engagée pour consolider ce constat ;
    \item construire la note technique complète du produit : garanties, bases
          techniques, méthodologie, tarification ;
    \item livrer un simulateur Excel opérationnel appliqué à une entreprise
          sénégalaise, alimenté par un questionnaire de collecte.
  \end{itemize}
\end{frame}

\begin{frame}{Méthode actuarielle : la méthode prospective}
  \vspace{0.2cm}
  Droits projetés au terme, probabilisés, actualisés, puis répartis au prorata
  de l'ancienneté.
  \vspace{0.25cm}
  \begin{block}{Salaire final et droits au terme}
    \vspace{-0.25cm}
    $$ S_f = \frac{SA}{12}\,(1+\tau_s)^{\max(60-x,\,0)}
       \qquad\qquad \text{Droits} = B(N_r)\times S_f $$
  \end{block}
  \begin{block}{Valeur actuelle probable des prestations futures}
    \vspace{-0.25cm}
    $$ \mathrm{VAP} = \text{Droits}\times\frac{l_{60}}{l_x}\times v^{\,n+1/2} $$
  \end{block}
  \small $x$ : âge ; $a$ : ancienneté ; $n$ : durée résiduelle ;
  $N_r = a + n$ : ancienneté au terme ; $v=(1+i)^{-1}$, $i=3{,}5\,\%$.
\end{frame}

\begin{frame}{De l'engagement au passif comptable}
  \vspace{0.2cm}
  \begin{block}{Dette actuarielle et charge normale}
    \vspace{-0.25cm}
    $$ \mathrm{DA} = \mathrm{VAP}\times\frac{a}{N_r}
       \qquad\qquad \mathrm{CN} = \frac{\mathrm{VAP}}{N_r} $$
  \end{block}
  \begin{block}{Passif de l'indemnité de licenciement}
    \vspace{-0.25cm}
    $$ \mathrm{Passif\ IL} = \mathrm{IL}\times\tau_\ell\times\frac{1-u^{\,n}}{1-u},
       \qquad u=\frac{1-\tau_r}{1+i} $$
  \end{block}
  \begin{block}{Tarification IFC+ (décès)}
    \vspace{-0.25cm}
    $$ \mathrm{PP} = \sum_j C_j\,q_{x_j}
       \qquad\qquad \mathrm{PC} = \frac{\mathrm{PP}}{1-g} $$
  \end{block}
\end{frame}

% ================================================================ SECTION 4
\section{Conclusion}

\begin{frame}{Conclusion}
  \vspace{0.3cm}
  \begin{itemize}\itemsep0.3cm
    \item les données FANAF établissent l'IFC, porté par la domination des
          assurances collectives (63,5\,\% des primes vie au Sénégal), comme
          le produit vie le plus commercialisé ; l'enquête élaborée s'inscrit
          dans cette dynamique de validation ;
    \item la méthode prospective fournit un passif social complet : engagement
          global de 230 M FCFA pour SENIA S.A., dette actuarielle de 111,1 M,
          charge normale de 8,2 M et passif de licenciement de 7,8 M ;
    \item trois modalités de financement au choix de l'entreprise et une
          couverture décès IFC+ à 0,35\,\% de la masse salariale ;
    \item trois livrables opérationnels : la note technique, le simulateur
          Excel et le questionnaire de collecte.
  \end{itemize}
\end{frame}
